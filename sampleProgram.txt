class HelloWorld {

    private var someInt: Int;
    private var someFloat: Float;
	
	init() {
		super();
	}

	fun main() {
		someInt = 0;
		someFloat = 13.2;
		
		while(someInt < 3) {
			someInt++;
			when(someInt) {
				0 -> someFloat = 13.0;
				1 -> someFloat = 13.1;
				else -> someFloat = 13.2;
			}
		}
		
	}
}
